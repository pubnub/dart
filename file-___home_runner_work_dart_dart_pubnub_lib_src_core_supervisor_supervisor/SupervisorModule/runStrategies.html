List<Resolution>? runStrategies(Fiber fiber, Diagnostic diagnostic) {
  return _strategies
      .map((strategy) => strategy.resolve(fiber, diagnostic))
      .firstWhere((resolutions) => resolutions != null, orElse: () => null);
}
